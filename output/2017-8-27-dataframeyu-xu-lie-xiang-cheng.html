def tile_df(df, n, m):
 dfn = df.T
 for _ in range(1, m):
 dfn = dfn.append(df.T, ignore_index=True)
 dfm = dfn.T
 for _ in range(1, n):
 dfm = dfm.append(dfn.T, ignore_index=True)
 return dfm

df = pandas.DataFrame([[1,2],[3,4]])
tile_df(df, 2, 3)
# 0 1 2 3 4 5
# 0 1 2 1 2 1 2
# 1 3 4 3 4 3 4
# 2 1 2 1 2 1 2
# 3 3 4 3 4 3 4
