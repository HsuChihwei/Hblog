用来计算一个人的体脂率

def get_BFR(weight, height, age, gender):
    """计算BMI及体脂率
    :param weight: 体重（kg）
    :param height: 身高（m）
    :param age: 年龄（岁）
    :param gender: 性别（男--1，女--0）
    :returns: 体脂率，float
    """
    # BMI
    BMI = weight /(height*height)
    # 体脂率 
    BFR = 1.2*BMI + 0.23*age - 5.4 - 10.8*gender
    return BFR

get_BFR(55, 1.7, 26, 1)
